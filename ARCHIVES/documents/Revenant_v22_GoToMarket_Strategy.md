**3. THE "RED BUTTON" SAFETY MECHANISM**

The Red Button UI is a critical safety control that prevents AI hallucinations from causing financial harm:

| Safety Layer | Function | Failure Mode Protection |
|--------------|----------|------------------------|
| **Schema Enforcement** | Only whitelisted tools allowed | Blocks "DELETE_USER_DATA" hallucinations |
| **Risk Scoring** | High-value = require approval | Prevents large unauthorized transfers |
| **Human Approval** | Risk Officer must click EXECUTE | No fully autonomous high-risk actions |
| **HMAC Verification** | Cryptographic proof of approval | Prevents forged approvals |
| **Idempotence Lock** | Same action cannot execute twice | Prevents double-spend |
| **Audit Trail** | Immutable record in Supabase | Regulatory compliance + forensics |

```javascript
// Tool Execution Engine — Core Logic
const TOOL_REGISTRY = {

  },

    endpoint: '/api/v1/auth/reset',
    risk_level: 'low',
    requires_approval: false,
    max_amount: null,
    allowed_params: ['user_id', 'delivery_method']
  },
  // FORBIDDEN TOOLS — Never executed
  DELETE_USER_DATA: { forbidden: true },
  TRANSFER_TO_EXTERNAL: { forbidden: true },
  MODIFY_KYC: { forbidden: true }
};

function executeTool(toolName, parameters, context) {
  // 1. Validate tool is allowed
  const tool = TOOL_REGISTRY[toolName];
  if (!tool || tool.forbidden) {
    throw new Error(`FORBIDDEN_TOOL: ${toolName} is not in the whitelist`);
  }
  
  // 2. Calculate risk score
  const riskScore = calculateRiskScore(tool, parameters, context);
  
  // 3. Route based on risk
  if (riskScore > 0.7 || tool.requires_approval) {
    return createApprovalRequest(toolName, parameters, riskScore);
  }
  
  // 4. Low-risk: Auto-execute
  return executeDirectly(tool, parameters);
}
```

```javascript
// Biometric Controller — Core Logic
const BIOMETRIC_CONFIG = {
  // Thresholds
  synthetic_marker_threshold: 0.3,
  high_value_threshold: 10000,      // $10,000 USD
  critical_value_threshold: 50000,  // $50,000 USD
  
  // Azure Speaker Recognition
  azure: {
    endpoint: 'https://westus.api.cognitive.microsoft.com',
    api_version: 'v2.0',
    verification_confidence_threshold: 0.5
  },

  enrollment: {
    required_samples: 3,
    min_duration_seconds: 3,
    min_snr_db: 20,
    waiting_period_days: 30  // For high-value actions
  }
};

function processVoiceAuthentication(audioPayload, userId, transactionAmount) {
  // 1. Check enrollment status
  const enrollment = getUserEnrollment(userId);
  if (!enrollment) {
    return {
      status: 'NOT_ENROLLED',
      action: 'TRIGGER_ENROLLMENT_FLOW',
      message: 'Voice ID not set up. Reply "Enroll Voice" to start.'
    };
  }
  
  // 2. Call Azure Speaker Recognition
  const azureResult = callAzureSpeakerAPI(audioPayload, enrollment.voiceprint_id);
  
  // 3. Detect synthetic markers
  const syntheticScore = detectSyntheticMarkers(audioPayload);
  
  // 4. Deepfake Protocol
  if (syntheticScore > BIOMETRIC_CONFIG.synthetic_marker_threshold) {
    sendSecurityAlert(userId, 'DEEPFAKE_DETECTED', syntheticScore);
    return {
      status: 'SPOOF_DETECTED',
      action: 'BLOCK',
      spoof_score: syntheticScore,
      message: 'Security alert: Voice authentication failed. Contact support.'
    };
  }
  
  // 5. Amount-based escalation
  if (transactionAmount >= BIOMETRIC_CONFIG.critical_value_threshold) {
    return {
      status: 'VERIFIED_BUT_INSUFFICIENT',
      azure_confidence: azureResult.confidence,
      required_next: 'VIDEO_CALL_WITH_AGENT',
      message: 'Voice verified. Video confirmation required for this amount.'
    };
  }
  
  // 6. Success
  return {
    status: 'VERIFIED',
    confidence: azureResult.confidence,
    action: 'PROCEED'
  };
}
```
